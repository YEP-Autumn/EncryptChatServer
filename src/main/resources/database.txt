CREATE TABLE MSG_TABLE
(
    FRIEND_ID     INTEGER(12) NOT NULL,
    USER_ID     INTEGER(12) NOT NULL,
    MSG      VARBINARY(4096),
    SEND_TIME TIMESTAMP
) ENGINE = INNODB
  DEFAULT CHARSET = UTF8MB4;

CREATE TABLE USER_INFO
(
    USER_ID      INTEGER(12)   NOT NULL UNIQUE PRIMARY KEY,
    M_KEY        VARBINARY(24) NOT NULL UNIQUE,
    FRIEND_ID    INTEGER(12),
    CONNECT_TIME TIMESTAMP,
    FRIEND_STATUS BOOLEAN
) ENGINE = INNODB
  DEFAULT CHARSET = UTF8MB4;
